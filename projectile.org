  chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
      if (request.action === \"saveOrg\") {
          chrome.storage.sync.get(null, (data) => {
              let uri;

              if (data.useNewStyleLinks) {
                  uri = \"org-protocol://\" +
                  data.protocol +
                      \"?template=\" +
                      data.template +
                      \"&url=\" +
                      request.url +
                      \"&title=\" +
                      request.title +
                      \"&body=\" +
                      request.content;
              } else {
                  uri = \"org-protocol://\" +
                  data.protocol +
                      \":/\" +
                      data.template +
                      \"/\" +
                      request.url +
                      \"/\" +
                      request.title +
                      \"/\" +
                      request.content;
              }

              location.href = uri;
          });
      }
  });
